// Fireworks const canvas = document.createElement("canvas"); document.body.appendChild(canvas); const ctx = canvas.getContext("2d"); function resize(){ canvas.width = window.innerWidth; canvas.height = window.innerHeight; } resize(); window.addEventListener("resize", resize); let particles = []; function firework(){ const x = Math.random() * canvas.width; const y = Math.random() * canvas.height / 2; for(let i=0;i<80;i++){ particles.push({ x:x, y:y, vx:Math.cos(i*Math.PI/40)*(2+Math.random()*3), vy:Math.sin(i*Math.PI/40)*(2+Math.random()*3), life:100 }); } } setInterval(firework,1500); function animate(){ ctx.clearRect(0,0,canvas.width,canvas.height); for(let i=particles.length-1;i>=0;i--){ let p=particles[i]; ctx.fillStyle=`hsla(${p.life*3},100%,60%,${p.life/100})`; ctx.beginPath(); ctx.arc(p.x,p.y,2,0,Math.PI*2); ctx.fill(); p.x+=p.vx; p.y+=p.vy; p.vy+=0.03; p.life--; if(p.life<=0){ particles.splice(i,1); } } requestAnimationFrame(animate); } animate();
